import com.google.gson.Gson;
import com.google.gson.JsonObject;
import com.google.gson.JsonParser;
import com.google.gson.reflect.TypeToken;
import com.ververica.cdc.connectors.mysql.source.MySqlSource;
import com.ververica.cdc.connectors.mysql.table.StartupOptions;
import com.ververica.cdc.debezium.JsonDebeziumDeserializationSchema;
import org.apache.flink.api.common.eventtime.WatermarkStrategy;
import org.apache.flink.connector.base.DeliveryGuarantee;
import org.apache.flink.connector.elasticsearch.sink.Elasticsearch7SinkBuilder;
import org.apache.flink.connector.elasticsearch.sink.ElasticsearchSink;
import org.apache.flink.connector.elasticsearch.sink.FlushBackoffType;
import org.apache.flink.streaming.api.environment.StreamExecutionEnvironment;
import org.apache.http.HttpHost;
import org.elasticsearch.action.index.IndexRequest;
import org.elasticsearch.client.Requests;

import java.lang.reflect.Type;
import java.time.LocalDate;
import java.util.Map;

public class FlinkMysqlApplication {

    public final static String MYSQL_SERVER = "192.168.30.100";
    public final static String ES_SERVER = "192.168.30.100";

    public static void main(String[] args) {

        try (StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment()) {
            MySqlSource<String> mySqlSource = MySqlSource.<String>builder()
                    .hostname(MYSQL_SERVER)
                    .port(3306)
                    .databaseList("flink") // set captured database, If you need to synchronize the whole database, Please set tableList to ".*".
                    .tableList("flink.book") // set captured table
                    .username("root")
                    .password("root")
                    .deserializer(new JsonDebeziumDeserializationSchema()) // converts SourceRecord to JSON String
                    // 是否全量读取应根据实际业务
                    .startupOptions(StartupOptions.initial())
                    .build();

            ElasticsearchSink<String> elasticsearchSink = new Elasticsearch7SinkBuilder<String>()
                    // 下面的设置使 sink 在接收每个元素之后立即提交，否则这些元素将被缓存起来
                    .setBulkFlushMaxActions(1)
                    .setDeliveryGuarantee(DeliveryGuarantee.AT_LEAST_ONCE)
                    .setHosts(new HttpHost(ES_SERVER, 9200, "http"))
                    .setEmitter((element, context, indexer) -> indexer.add(createIndexRequest(element)))
                    // 这里启用了一个指数退避重试策略，初始延迟为 1000 毫秒且最大重试次数为 5
                    .setBulkFlushBackoffStrategy(FlushBackoffType.EXPONENTIAL, 3, 1000)
                    .build();

            // env
            //         // 每3000毫秒执行一次checkpoint
            //         .enableCheckpointing(3000)
            //         .getCheckpointConfig()
            //         .setCheckpointStorage("file:///opt/flink/storage");

            env
                    .setParallelism(1)
                    .fromSource(mySqlSource, WatermarkStrategy.noWatermarks(), "MySQL Source")
                    .sinkTo(elasticsearchSink);
                    // .print("==>");

            env.execute("Print MySQL Binlog");
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }

    private static IndexRequest createIndexRequest(String element) {
        JsonObject jsonObject = JsonParser.parseString(element).getAsJsonObject();
        JsonObject after = jsonObject.getAsJsonObject("after");
        // id
        String id = after.get("id").getAsString();
        // map
        Type type = new TypeToken<Map<String, Object>>() {
        }.getType();

        Gson gson = new Gson();
        Map<String, Object> map = gson.fromJson(after, type);

        return Requests.indexRequest()
                .index("book")
                .id(id)
                .source(map);
    }
}

# 3.部署

# 1.导出依赖

  1. idea双击shift输入run anything
  2. 在新窗口输入: mvn dependency:copy-dependencies -DoutputDirectory=C:\Users\heartape\Desktop\flink -DstripVersion=true
  3. 删除关于log4j的依赖

# 2.部署flink(docker单机)

# 启动容器
docker run -d -t --name flink -p 8081:8081 -e JOB_MANAGER_RPC_ADDRESS=localhost -e "TZ=Asia/Shanghai" flink:1.16.1 jobmanager
# 进入容器
docker exec -it flink bash

# 拉取依赖
apt-get update
# ping用于调试网络
apt install iputils-ping
# ssh
apt-get -y install openssh-client
scp -r xxx@192.168.31.5:/C:/Users/heartape/Desktop/flink/* /opt/flink/lib
# 启动task,在job内启动
./bin/start-cluster.sh

# 如果是集群需要创建网络
docker network create -d bridge flink-bridge
docker network ls
# 查看容器网络
docker network inspect flink-bridge

# 4.测试

打开192.168.30.100:8001，上传任务即可。

# 5.注意点

  1. windows默认序列化不是utf-8，使用flink-cdc提供的json反序列化在windows测试会有乱码，linux正常。
  2. 上传任务选择没有依赖的任务，有依赖的仅用于测试，否则在运行时会有依赖冲突，同时也能减小上传任务时间和服务器存储压力。
  3. 以上步骤均为开发环境步骤，正式环境应注意高可用。

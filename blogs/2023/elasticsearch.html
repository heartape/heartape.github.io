# 1.简介

elasticsearch是目前使用最为广泛的中间件之一，在搜索领域有着难以替代的地位。 可以通过rest客户端的方式和es进行交互，所以兼容性非常强。

# 2.nested

nested嵌套文档会将从表数据保存到主表中，这种连接查询的方式性能较强，结构清晰，可读性强。 但是不适合修改较多或从表与主表的从属关系不明确（从表可能属于多个主表）的场景。

# 创建索引(book数据保存到bookshelf的文档中)

PUT /bookshelf
{
    "mappings": {
        "dynamic": "strict",
        "properties": {
            "id": {
                "type": "keyword",
                "index": "false"
            },
            "no": {
                "type": "keyword"
            },
            "book": {
                // 关键字会将该字段视为从表数据，否则搜索时会视为一个普通字段
                "type": "nested",
                "properties": {
                    "id": {
                        "type": "keyword",
                        "index": "false"
                    },
                    "bookshelf_id": {
                        "type": "keyword",
                        "index": "false"
                    },
                    "title": {
                        // text插入分词，配合term搜索不分词，可以部分匹配，配合match搜索分词，可以全文搜索
                        "type": "text"
                    },
                    "author": {
                        // keyword插入不分词，配合term搜索不分词，可以完全匹配
                        "type": "keyword"
                    },
                    "price": {
                        "type": "integer"
                    },
                    "publish_date": {
                        "type": "date",
                        "format": "strict_date || epoch_millis"
                    }
                }
            }
        }
    }
}

# 3.join

针对nested某些情况下不够灵活且数据冗余的问题，join或许是不错的方案。 原理就是将多个索引的数据存到一个索引里，通过join字段来区分。 官方文档也提到join相比nested性能相差较多，并且数据可读性非常差，需要按实际情况取舍。

# 创建bookshelf索引（相当于同时也创建了book索引）

PUT /bookshelf
{
    "mappings": {
        "properties": {
            "id": {
                "type": "keyword",
                "index": "false"
            },
            "no": {
                "type": "keyword"
            },
            "master_slave": {
                "type": "join",
                "relations": { 
                    // 定义了文档内的一组可能的关系，每个关系是父名和子名，可以定义多级join
                    "master": "slave"
                }
            },
            // 下面是从表数据
            "book_id": {
                "type": "keyword",
                "index": "false"
            },
            "bookshelf_id": {
                "type": "keyword",
                "index": "false"
            },
            "title": {
                "type": "text"
            },
            "author": {
                "type": "keyword"
            },
            "price": {
                "type": "integer",
                "index": "false"
            },
            "publish_date": {
                "type": "date",
                "format": "strict_date || epoch_millis"
            }
        }
    }
}

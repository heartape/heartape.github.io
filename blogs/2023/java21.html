# 1.VirtualThread

VirtualThread在创建时不必向操作系统申请资源，而是使用ForkJoinPool的任务窃取机制，使用一个平台线程来管理多个VirtualThread。 这种特性表明如果在存在io任务的阻塞任务时，将充分利用平台线程。 但如果执行的是计算密集非阻塞任务时，VirtualThread反而会有更高的成本。

在VirtualThread后就再也不必使用线程池了，因为它的创建和销毁很轻量。

class Solution {
    /**
     * VirtualThread的基本api
     */
    public static void virtualThread() {
        Thread.ofVirtual().unstarted(() -> {
            // todo
        }).start();

        Thread.ofVirtual().start(() -> {
            // todo
        });

        ThreadFactory threadFactory = Thread.ofVirtual().name("virtual").factory();
        threadFactory.newThread(() -> {
            // todo
        });

        try (ExecutorService executorService = Executors.newVirtualThreadPerTaskExecutor()) {
            executorService.submit(() -> {
                // todo
            });
        }
    }
}

final class VirtualThread extends BaseVirtualThread {
    
    private void run(Runnable task) {
        assert state == RUNNING;
        // first mount
        mount();
        notifyJvmtiStart();
        // 绑定时已经继承ScopedValue了
        Object bindings = scopedValueBindings();
        try {
            runWith(bindings, task);
        } catch (Throwable exc) {
            dispatchUncaughtException(exc);
        } finally {
            try {
                // pop any remaining scopes from the stack, this may block
                StackableScope.popAll();
            } finally {
                // last unmount
                notifyJvmtiEnd();
                // 卸载虚拟线程
                unmount();
                // final state
                setState(TERMINATED);
            }
        }
    }

    /**
     * 装载虚拟线程
     */
    private void mount() {
        // sets the carrier thread
        // 通过载体线程来传递
        Thread carrier = Thread.currentCarrierThread();
        // 将平台线程和虚拟线程互相绑定
        setCarrierThread(carrier);
        carrier.setCurrentThread(this);
    }
}

# 2.StructuredTaskScope

class Solution {
    /**
     * StructuredTaskScope不需要使用者单独获取每一个线程的结果，只要执行成功，其他线程自动停止，并通过join返回值。
     * 同时默认ThreadFactory为虚拟线程工厂
     * <pre>
     *     {@code
     *          public ShutdownOnSuccess() {
     *             this(null, Thread.ofVirtual().factory());
     *         }
     *     }
     * </pre>
     */
    public static void success() {
        try (StructuredTaskScope.ShutdownOnSuccess<String> taskScope = new StructuredTaskScope.ShutdownOnSuccess<String>()) {
            // StructuredTaskScope.Subtask<String> task = taskScope.fork("1"); 获得的是一个技术对象，不建议使用。
            // 此时启动的线程是VirtualThread
            taskScope.fork(() -> "1");
            taskScope.fork(() -> "2");
            taskScope.fork(() -> "3");
            // 第一个执行成功的线程的结果
            String result = taskScope.join().result();
            System.out.println(result);
        } catch (ExecutionException | InterruptedException e) {
            throw new RuntimeException(e);
        }
    }

    /**
     * StructuredTaskScope不需要使用者单独获取每一个线程的结果，只要执行失败，其他线程自动停止
     */
    public static void failure() {
        try (StructuredTaskScope.ShutdownOnFailure taskScope = new StructuredTaskScope.ShutdownOnFailure()) {
            StructuredTaskScope.Subtask<String> subtask1 = taskScope.fork(() -> "1");
            StructuredTaskScope.Subtask<String> subtask2 = taskScope.fork(() -> "2");
            StructuredTaskScope.Subtask<String> subtask3 = taskScope.fork(() -> "3");

            taskScope.join();
            taskScope.throwIfFailed(Function.identity());

            String result = Stream.of(subtask1, subtask2, subtask3)
                    .map(Supplier::get)
                    .collect(Collectors.joining(", ", "{ ", " }"));
            System.out.println(result);
        } catch (Throwable e) {
            throw new RuntimeException(e);
        }
    }
}

# 3.ScopedValue

class Solution {
    
    private final static ScopedValue<String> CONTEXT = ScopedValue.newInstance();

    /**
     * 绑定的值只能再run的作用域内使用，覆盖的话也只会在作用域内覆盖，使用时不需要像ThreadLocal那样关注释放。
     */
    public static void value() {
        ScopedValue.where(CONTEXT, "a")
                .run(() -> {
                    System.out.println("1:" + CONTEXT.get());
                    ScopedValue.where(CONTEXT, "b")
                            .run(() -> System.out.println("2:" + CONTEXT.get()));
                    System.out.println("3:" + CONTEXT.get());
                });
    }
}

class Solution {
    
    private final static ScopedValue<String> CONTEXT = ScopedValue.newInstance();

    /**
     * 与StructuredTaskScope配合可以轻松实现线程间变量共享,同时也不用像静态变量那样担心线程安全。
     * 同时也表明ScopedValue是可以被子线程继承的
     */
    public static void task() {
        ScopedValue.runWhere(CONTEXT, "hello", () -> {
            try (StructuredTaskScope.ShutdownOnSuccess<String> taskScope = new StructuredTaskScope.ShutdownOnSuccess<String>()) {
                taskScope.fork(CONTEXT::get);
                taskScope.fork(CONTEXT::get);
                taskScope.fork(CONTEXT::get);

                String result = taskScope.join().result();
                System.out.println(result);
            } catch (ExecutionException | InterruptedException e) {
                throw new RuntimeException(e);
            }
        });
    }
}

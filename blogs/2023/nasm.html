# 1.环境搭建

工欲善其事，必先利其器。

# 1.软件安装

  1. Bochs
  2. nasm
  3. dd
  4. sublime(vscode)

都是基于windows平台的环境配置，centos nasm安装方式如下：

yum update
yum install make
yum install gcc gcc-c++ -y

wget --no-check-certificate https://www.nasm.us/pub/nasm/releasebuilds/2.16/nasm-2.16.tar.gz
tar zxvf nasm-2.16.tar.gz
cd nasm-2.16.tar.gz
./configure
make install

# 2.脚本配置

同样的也是配置sublime脚本来编译。

配置sublime的步骤：

  1. 安装代码美化插件(略)
  2. 配置编译菜单
  3. 编写编译脚本

配置编译菜单详细做法(sublime)：

  • tools -> build system -> new build system
  • 输入以下代码并保存文件名为 Assembly.sublime-build
  • 文件在路径 C:\Users\heartape\AppData\Roaming\Sublime Text 3\Packages\User
{
    "cmd": ["\"选择其他编译选项\""],
    "file_regex": "^[ ]*File \"(...*?)\", line ([0-9]*)",
    "selector": "source.asm",
    "encoding":"cp936",
    "variants":  
     [
        {
         "name": "nasm-img-init",
         "cmd": ["D:\\nasm\\nasm-img-init.bat", "$file_base_name"],
         "shell":true
        },
        {
         "name": "nasm-img-run",
         "cmd": ["D:\\nasm\\nasm-img-run.bat", "$file_base_name"],
         "shell":true
        }
     ]  
}

# 2.引导扇区

# 1.示例代码

最简单的引导扇区代码

boot.asm

    jmp near start
         
hello db 'hello world!'

start:
	mov ax,7c0h
	mov ds,ax
	xor si,si

	mov ax,0b800h
	mov es,ax
	xor di,di

	mov cx,12
            
show:
	mov al,[si+3]
	mov ah,4
	mov [es:di],ax
	add di,2
	inc si
	loop show

	jmp near ($) ;循环

  	times 510 - ($ - $$) db 0 ;计算需要填充的空格
    db 55h, 0aah ;必须已55aa结尾

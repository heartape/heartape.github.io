# 2.负载均衡

# 1.轮询(默认)

#在http块内,server块外定义多台服务器
upstream myserver{
	server 192.168.31.102:8080;
	server 192.168.31.103:8080;
}
#location下添加代理配置
location / {
	#nginx代理给其他ip时会丢失请求的host信息导致无法访问,proxy_set_header重新进行设置
	proxy_set_header Host $host;
	proxy_pass http://myserver;
}

global_defs {
	notification_email {
	  acassen@firewall.loc
	  failover@firewall.loc
	  sysadmin@firewall.loc
	}
	notification_email_from Alexandre.Cassen@firewall.loc
	smtp_ server 192.168.17.129

vrrp_script chk_http_ port {
	//检测脚本路径
	script "/usr/local/src/nginx_check.sh"
	interval 2   # (检测脚本执行的间隔)2s
	weight 2  #权重，如果这个脚本检测为真，服务器权重+2
}

vrrp_instance VI_1 {
	state MASTER   # 备份服务器上将MASTER 改为BACKUP
	interface ens33 //网卡名称
	virtual_router_id 51 # 主、备机的virtual_router_id必须相同
	priority 100   #主、备机取不同的优先级，主机值较大，备份机值较小
	advert_int 1	#每隔1s发送一次心跳
	authentication {	# 校验方式， 类型是密码，密码1111
        auth type PASS
        auth pass 1111
    }
	virtual_ipaddress { # 虛拟ip
		192.168.17.50 // VRRP H虛拟ip地址
	}
}

if [ $A -eq 0];then
	/usr/local/nginx/sbin/nginx
	sleep 2
	if [`ps -C nginx --no-header| wc -1` -eq 0 ];then
		killall keepalived
	fi
fi

# 5.配置https

listen       443 ssl;
listen  [::]:443 ssl;
server_name  blog.heartape.com;

ssl_certificate https/blog.heartape.com_nginx/blog.heartape.com.pem(证书相对路径);
ssl_certificate_key https/blog.heartape.com_nginx/blog.heartape.com.key(证书相对路径);
ssl_session_timeout 5m;
ssl_protocols TLSv1 TLSv1.1 TLSv1.2;
ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:HIGH:!aNULL:!MD5:!RC4:!DHE;
ssl_prefer_server_ciphers on;

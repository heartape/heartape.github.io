# 1.CompletableFuture

在1.5时就有了Future接口，后来在1.8时增加了CompletionStage接口，CompletableFuture就是针对Future的功能增强

import java.util.concurrent.CompletableFuture;
import java.util.concurrent.Executors;
import java.util.concurrent.ThreadPoolExecutor;
import java.util.concurrent.TimeUnit;

class Test {
    public static void main(String[] args) {
        // CompletableFuture默认线程为守护线程，会在主线程结束后立即结束
        ThreadPoolExecutor threadPoolExecutor = (ThreadPoolExecutor) Executors.newFixedThreadPool(10);
        CompletableFuture<String> future = CompletableFuture
                // 开启有返回值
                .supplyAsync(() -> "1", threadPoolExecutor)
                // 开启无返回值
                // .runAsync(() -> System.out.println("1"))
                .thenApply(value -> value + "2")
                .handle((value, e) -> value + "3")
                // .thenAccept(System.out::println)
                .whenComplete((value, e) -> value + "4")
                .exceptionally(e -> "出错了");
        System.out.println("main等待");
        // 获取结果
        boolean complete = future.complete("等待超时了");
        // String result = future.get();
        String result = future.get(1, TimeUnit.SECONDS);
        // String result = future.join();
        // String result = future.getNow("等待超时了");
    }
}

# 2.LockSupport

线程等待/唤醒机制有3种

  • synchronized + wait() + notify()
  • ReentrantLock + Condition + await() + signal()
  • LockSupport.park() + LockSupport.unpark(Thread)

这里有一个考点:Thread.sleep(),LockSupport.park()不会释放占有的锁，Object.wait(),Condition.await()会释放占有的锁;

LockSupport与前两种不同的是:

  • 不需要显式地获取锁
  • 以令牌的形式存储唤醒指令，且最多存储一个令牌，每次唤醒消耗一个令牌
  • 唤醒指令可以在等待指令前发出，使得程序不会因为提前发出唤醒指令导致程序阻塞

# 3.volatile

volatile是一个关键词，主要功能就是强制刷新线程私有内存，相当于直接从主内存读取数据。特点:

  • 保证可见性
  • 不保证原子性
  • 禁止指令重排

volatile的底层实现就是内存屏障，字节码会在volatile变量操作指令(ACC_VOLATILE)前后添加内存屏障(StoreStore,StoreLoad,LoadStore,LoadLoad)。 内存屏障的功能:

  • 阻止屏障两边指令重排(禁止将volatile写操作之前的操作排到后面，禁止将volatile读操作之后的操作排到前面)
  • 写操作强制将数据刷回主内存
  • 读操作操作从主内存获取数据

volatile和synchronized可能会混淆:

  • volatile会刷新变量，synchronized则是锁定当前变量
  • volatile可见性(不要使用自增等非原子操作)，synchronized可见性和原子性
  • volatile不会造成线程阻塞，synchronized会造成线程阻塞

# 5.LongAdder和LongAccumulator(累加器)

相对于AtomicLong优化了性能，将少了自旋，高争用的情况下性能提升明显，但会加大内存消耗。 LongAdder只能用于递增，且只能从0开始；LongAccumulator可以自定义计算公式和初始值。

# 6.ThreadLocal

ThreadLocal是线程内的局部变量，这种变量在线程的生命周期内起作用，线程间相互隔离

# 2.原理

Thread内维护了一个ThreadLocalMap，所以线程间隔离，避免了并发时锁的竞争

ThreadLocal.ThreadLocalMap threadLocals = null;

我这里就分析一下ThreadLocal的get()方法，set()和remove()基本思想一致

public class ThreadLocal<T> {
    // 用于ThreadLocalMap散列计算
    private final int threadLocalHashCode = nextHashCode();
    // 通过原子类保证了线程间threadLocalHashCode不同
    private static AtomicInteger nextHashCode = new AtomicInteger();
    
    public T get() {
        Thread t = Thread.currentThread();
        ThreadLocalMap map = getMap(t);
        if (map != null) {
            // ThreadLocal对象为k通过hash算法取值
            ThreadLocalMap.Entry e = map.getEntry(this);
            if (e != null) {
                // value获取值，get()获取key
                T result = (T)e.value;
                return result;
            }
        }
        return setInitialValue();
    }

    ThreadLocalMap getMap(Thread t) {
        // 每个线程都维护了一个ThreadLocalMap
        return t.threadLocals;
    }

    static class ThreadLocalMap {

        private Entry[] table;

        private int size = 0;

        private static int nextIndex(int i, int len) {
            // 返回下一个index，超过数组范围后从0开始
            return ((i + 1 < len) ? i + 1 : 0);
        }
        
        private Entry getEntry(ThreadLocal<?> key) {
            // 计算散列位置
            int i = key.threadLocalHashCode & (table.length - 1);
            Entry e = table[i];
            if (e != null && e.get() == key)
                return e;
            else
                // 没找到可能是碰撞后放到了其他位置
                return getEntryAfterMiss(key, i, e);
        }

        private Entry getEntryAfterMiss(ThreadLocal<?> key, int i, Entry e) {
            Entry[] tab = table;
            int len = tab.length;
            // Entry为空表明数据已被清除
            while (e != null) {
                ThreadLocal<?> k = e.get();
                // 找到目标
                if (k == key)
                    return e;
                // ThreadLocal为空表明可能已经被回收
                // 有人可能会有疑问，ThreadLocal被table作为弱引用持有，会不会被提前gc
                // 其实ThreadLocal在执行没结束时同时还被强引用了，所以不会被提前gc
                // 如果被static修饰更不可能被提前gc
                if (k == null)
                    // 删除失效引用并重新散列
                    expungeStaleEntry(i);
                else
                    // 循环获取空闲的index
                    i = nextIndex(i, len);
                e = tab[i];
            }
            return null;
        }

        private int expungeStaleEntry(int staleSlot) {
            Entry[] tab = table;
            int len = tab.length;
            // 删除ThreadLocal为空的Entry
            // 用来自动清理内存泄漏，我们在使用时可以手动remove来清除
            tab[staleSlot].value = null;
            tab[staleSlot] = null;
            size--;
            // 重新散列
            Entry e;
            int i;
            // 循环下一个空闲位置
            for (i = nextIndex(staleSlot, len);
                 (e = tab[i]) != null;
                 i = nextIndex(i, len)) {
                ThreadLocal<?> k = e.get();
                if (k == null) {
                    // 删除ThreadLocal为空的Entry
                    e.value = null;
                    tab[i] = null;
                    size--;
                } else {
                    // 计算散列位置
                    int h = k.threadLocalHashCode & (len - 1);
                    // 如果散列位置与当前位置不符合
                    if (h != i) {
                        // 清除当前位置数据
                        tab[i] = null;
                        // != null说明有数据，寻找没有数据的位置
                        while (tab[h] != null)
                            h = nextIndex(h, len);
                        // 将数据放到新的位置
                        tab[h] = e;
                    }
                }
            }
            return i;
        }
    }
}

# 3.强软弱虚

强引用:正常的对象引用。

软引用(SoftReference<>):内存不足时会进行清除。

弱引用(WeakReference<>):每次gc都会进行清除。

虚引用(PhantomReference<>):必须配合(ReferenceQueue<>)使用。 无法通过虚引用来获取一个对象的实例。 每次gc都会进行清除。 设置虚引用关联的唯一目的就是能在这个对象被收集器回收时收到一个系统通知，一个很重要的用途就是这个通过通知来实现堆外内存的释放。

# 1.对象内存布局

java运行时内存包含一下结构:

  • Native Method Stack(本地方法栈)
  • VM Stack(虚拟机栈)
  • Project Counter Register(程序计数器)
  • Heap(堆)
  • Metaspace(元空间，即方法区)

我们在调优时主要针对的就是堆，而保存对象的区域也是堆，jvm堆内存包括以下分区:

  • eden(伊甸):保存新创建的对象，如果对象太大或eden内存不足也等可能直接存入老年区
  • s0,s1(幸存):为了内存完整，幸存对象会被来回拷贝，最多15次(默认)会被存入老年区
  • old(老年)

对象内存布局:

  • 对象头
    • Mark Word(对象标记):
    • Class Pointer类元信息(类型指针):指向类信息首地址
    • Length(仅数组)
  • 实例数据
  • 对齐填充:保证8个字节的倍数，利于检索

# 2.锁升级阶段

鉴于重量级锁会导致用户态和内核态的上下文切换，以此造成巨大的开销，产生的锁的升级机制。

无锁:jvm会有4秒的偏向锁开启的延迟时间，此时对象处于001无锁态(无锁不可偏向状态)。如果关闭偏向锁启动延迟、或是经过4秒且没有线程竞争对象的锁，那么对象会进入101无锁可偏向状态(匿名偏向)。 在无锁不可偏向状态下，如果有线程试图获取锁，那么将跳过升级偏向锁的过程，直接使用轻量级锁。 如果先获取监视器的hashCode()，会直接使对象回到无锁态，加锁时会使用轻量级锁。

注意:如果没有使用到hashCode()，那么Mark Word不会保存hashCode。轻量级锁和重量级锁将hashCode保存在锁指针中。

偏向锁:第一个试图获取该对象的锁的线程，会使用CAS操作尝试将自己的threadID写入对象头的mark word中，使匿名偏向状态升级为已偏向的偏向锁状态。 如果之后有线程再次尝试获取锁时，需要检查mark word中存储的threadID是否与自己相同即可。 如果threadID与自己不同，但监视器处于匿名偏向状态或可重偏向状态时也可以写入threadID。否则将升级锁。 如果先加锁，再获取监视器的hashCode()，会直接使用重量级锁。另外wait()也会导致升级为重量级锁，因为它依赖于monitor。

注意:偏向锁在jdk15弃用。

轻量级锁:在代码访问同步资源时，如果锁对象处于无锁不可偏向状态，jvm首先将在当前线程的栈帧中创建一条锁记录。 jvm使用CAS操作尝试将对象的轻量级锁指针栈帧中的锁记录，CAS替换失败一定次数后将升级为重量级锁。

重量级锁:依赖对象内部的monitor(监视器或者说管程)来实现,而monitor又依赖于操作系统底层的Mutex Lock(互斥锁)实现，会导致用户态和内核态的上下文切换。

# 8.aqs

AbstractQueuedSynchronizer本身源码比较复杂且零散，直接直接分析一头雾水，所以就从其子类ReentrantLock的非公平实现来进行分析。

# 1.加锁

加锁部分较为复杂，包含了初始化队列和清楚队列等功能。

final void lock() {
    if (!initialTryLock())
        acquire(1);
}
// 初始化
final boolean initialTryLock() {
    Thread current = Thread.currentThread();
    // first attempt is unguarded(cas操作)
    if (compareAndSetState(0, 1)) {
        // 如果为0代表锁空闲
        setExclusiveOwnerThread(current);
        return true;
    } 
    // 是否是当前线程占有锁
    else if (getExclusiveOwnerThread() == current) {
        // 因为ReentrantLock是可重入锁，state + 1
        int c = getState() + 1;
        if (c < 0) // overflow // int类型溢出?没找到对其进行小于0的赋值操作
            throw new Error("Maximum lock count exceeded");
        setState(c);
        return true;
    } else
        return false;
}

public final void acquire(int arg) {
    if (!tryAcquire(arg))
        // 第二次抢占锁不成功
        acquire(null, arg, false, false, false, 0L);
}

protected final boolean tryAcquire(int acquires) {
    // 锁空闲,cas操作
    if (getState() == 0 && compareAndSetState(0, acquires)) {
        setExclusiveOwnerThread(Thread.currentThread());
        return true;
    }
    return false;
}

final int acquire(Node node, int arg, boolean shared, boolean interruptible, boolean timed, long time) {
    Thread current = Thread.currentThread();
    byte spins = 0, postSpins = 0;
    //first是否是第一个节点，这里的第一个是指head的下一个，head是哨兵节点
    boolean interrupted = false, first = false;
    Node pred = null;

    for (;;) {
        // 将node的前一个节点赋值给pred
        if (!first && (pred = (node == null) ? null : node.prev) != null && !(first = (head == pred))) {
            if (pred.status < 0) {
                // repeatedly traverses from tail, unsplicing cancelled nodes until none are found
                cleanQueue();           // predecessor cancelled
                continue;
            } else if (pred.prev == null) {
                Thread.onSpinWait();    // ensure serialization //旋转等待,虚拟机"可能"会对该操作进行性能优化
                continue;
            }
        }
        // 无法向前遍历
        if (first || pred == null) {
            boolean acquired;
            try {
                // false
                if (shared)
                    acquired = (tryAcquireShared(arg) >= 0);
                else
                    // 再次尝试
                    acquired = tryAcquire(arg);
            } catch (Throwable ex) {
                cancelAcquire(node, interrupted, false);
                throw ex;
            }
            if (acquired) {
                // 尝试成功
                if (first) {
                    // 设置首节点
                    node.prev = null;
                    head = node;
                    pred.next = null;
                    node.waiter = null;
                    if (shared)
                        signalNextIfShared(node);
                    // 希望进行中断
                    if (interrupted)
                        // 设置中断标志
                        current.interrupt();
                }
                return 1;
            }
        }
        if (node == null) {
            // 如果当前节点为空就新建一个节点
            node = new ExclusiveNode();
        } else if (pred == null) {
            // 如果没有前一个节点，就将当前线程赋给该节点
            node.waiter = current;
            Node t = tail;
            // 设置prev属性为尾节点，放到队列末尾
            node.setPrevRelaxed(t);
            if (t == null)
                // 没有尾节点说明队列还未初始化
                tryInitializeHead();
            else if (!casTail(t, node))
                // cas修改尾节点失败的话，就清除当前节点的前节点，继续循环修改
                node.setPrevRelaxed(null);
            else
                // 修改成功
                t.next = node;
        } else if (first && spins != 0) {
            // 如果是首节点则空转一次(意图未知)
            --spins;
            Thread.onSpinWait();
        } else if (node.status == 0) {
            /**
             * java8中的waitStatus:
             * static final int CANCELLED =  1;
             * static final int SIGNAL    = -1;
             * static final int CONDITION = -2;
             * static final int PROPAGATE = -3;
             * 
             * java17中的status:
             * static final int WAITING   = 1;
             * static final int CANCELLED = 0x80000000;
             * static final int COND      = 2; 
             * 把节点状态改为 WAITING才能park，CANCELLED才能unpark
             */
            node.status = WAITING;
        } else {
            // 先左移(乘2)，再或(加1)，并重置剩余可以空转的次数，postSpins表示总共可以空转的次数，spins表示剩余可以空转的次数。
            spins = postSpins = (byte)((postSpins << 1) | 1);
            // timed == false
            if (!timed)
                // 没有超时设置的阻塞
                LockSupport.park(this);
            // 如果interrupted标记状态与当前线程中断状态不同(线程中断状态改变)就将标记状态改为中断。interruptible总是不可中断
            if ((interrupted |= Thread.interrupted()) && interruptible)
                break;
        }
    }
}

# 2.解锁

解锁较为简单，本质上就是LockSupport.unpark()

public class ReentrantLock {
  public void unlock() {
    sync.release(1);
  }

  protected final boolean tryRelease(int releases) {
    int c = getState() - releases;
    // 是否独占线程
    if (getExclusiveOwnerThread() != Thread.currentThread())
      throw new IllegalMonitorStateException();
    // 是否空闲，因为还未解锁，所以不可能空闲
    boolean free = (c == 0);
    if (free)
      setExclusiveOwnerThread(null);
    // cas
    setState(c);
    return free;
  }
}

public abstract class AbstractQueuedSynchronizer {
  public final boolean release(int arg) {
    if (tryRelease(arg)) {
      signalNext(head);
      return true;
    }
    return false;
  }

  private static void signalNext(Node h) {
    Node s;
    // 下一节点不为空
    if (h != null && (s = h.next) != null && s.status != 0) {
      // 取消等待设置。WAITING=1，将状态末位置为0(CANCELLED)，但当status为2(COND)时,末位置为0不受影响，很巧妙。
      s.getAndUnsetStatus(WAITING);
      // 成功修改后解锁
      LockSupport.unpark(s.waiter);
    }
  }
}

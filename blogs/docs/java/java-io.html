class Test {
    /**
     * 写入文件
     */
    private static void writeText(String fileName) throws IOException {
        File file = new File(fileName);
        FileWriter fileWriter = new FileWriter(file);
        BufferedWriter bufferedWriter = new BufferedWriter(fileWriter);
        bufferedWriter.write("三点钟了，");
        bufferedWriter.flush();
        char[] chars = {'饮','茶','先','了','。'};
        bufferedWriter.write(chars);
        bufferedWriter.flush();
        bufferedWriter.close();
    }

    /**
     * 删除文字
     * @param fileName 原文件名
     * @param fileCopyName 复制文件名，作为中间载体，需要验证源路径下不存在以免覆盖现有文件
     * @param offset 行数
     * @param start 目标行删除开始位置
     * @param end 目标行删除结束位置
     */
    private static void deleteText(String fileName, String fileCopyName, int offset, int start, int end) throws IOException {
        File file = new File(fileName);
        File fileCopy = new File(fileCopyName);
        FileReader fileReader = new FileReader(file);
        FileWriter fileWriter = new FileWriter(fileCopy);
        BufferedReader bufferedReader = new BufferedReader(fileReader);
        BufferedWriter bufferedWriter = new BufferedWriter(fileWriter);
        String s;
        int number = 0;
        while ((s = bufferedReader.readLine()) != null) {
            number++;
            if (number == offset) {
                s = s.substring(0,start) + s.substring(end + 1);
            }
            bufferedWriter.write(s + "\n");
        }
        bufferedWriter.flush();
        bufferedReader.close();
        bufferedWriter.close();

        if (file.delete()) {
            boolean b = fileCopy.renameTo(file);
            System.out.println(b ? "删除成功" : "删除失败");
        }
    }

    /**
     * 字符流复制文本
     */
    private static void copyText(String fileName, String fileCopy) throws IOException {
        File file = new File(fileName);
        File file1 = new File(fileCopy);
        FileReader fileReader = new FileReader(file);
        FileWriter fileWriter = new FileWriter(file1);
        BufferedReader bufferedReader = new BufferedReader(fileReader);
        BufferedWriter bufferedWriter = new BufferedWriter(fileWriter);
        char[] chars = new char[10];
        int length;
        while ((length = bufferedReader.read(chars)) > -1) {
            // String s = bufferedReader.readLine();
            bufferedWriter.write(chars, 0, length);
            bufferedWriter.flush();
        }
        bufferedReader.close();
        bufferedWriter.close();
    }

    /**
     * 字节流复制文件
     */
    private static void copyFile(String fileName, String fileCopy) throws IOException {
        File file = new File(fileName);
        File file1 = new File(fileCopy);
        InputStream inputStream = new FileInputStream(file);
        BufferedInputStream bufferedInputStream = new BufferedInputStream(inputStream);
        FileOutputStream fileOutputStream = new FileOutputStream(file1);
        BufferedOutputStream bufferedOutputStream = new BufferedOutputStream(fileOutputStream);

        byte[] bytes = new byte[10];
        int length;
        while ((length = bufferedInputStream.read(bytes)) > -1) {
            bufferedOutputStream.write(bytes,0, length);
            bufferedOutputStream.flush();
        }
        bufferedOutputStream.close();
        bufferedInputStream.close();
    }

    /**
     * http下载文件
     */
    private static void download(String fileName, String fileCopy) throws IOException {
        // http地址
        URL url = new URL(fileName);
        HttpURLConnection connection = (HttpURLConnection)url.openConnection();
        connection.setRequestMethod("GET");
        connection.connect();
        InputStream inputStream = connection.getInputStream();
        BufferedInputStream bufferedInputStream = new BufferedInputStream(inputStream);

        File file = new File(fileCopy);
        FileOutputStream fileOutputStream = new FileOutputStream(file);
        BufferedOutputStream bufferedOutputStream = new BufferedOutputStream(fileOutputStream);

        byte[] bytes = new byte[10];
        int length;
        while ((length = bufferedInputStream.read(bytes)) > -1) {
            bufferedOutputStream.write(bytes,0, length);
            bufferedOutputStream.flush();
        }
        bufferedOutputStream.close();
        bufferedInputStream.close();
    }
}

# 3.实现简单数据传输

import java.io.*;
import java.net.*;
import java.nio.charset.StandardCharsets;

public class Connect {

    private final Socket socket;
    private BufferedReader bufferedReader;
    private BufferedOutputStream bufferedOutputStream;

    public Connect(String address, Integer port, Integer localPort) throws IOException {
        InetAddress inet4Address = Inet4Address.getByName(address);
        InetAddress localAddress = Inet4Address.getByName("127.0.0.1");
        socket = new Socket(inet4Address, port, localAddress, localPort);
    }

    public Connect(Socket socket) {
        this.socket = socket;
    }

    public boolean start() {
        InputStream inputStream;
        OutputStream outputStream;
        try {
            inputStream = socket.getInputStream();
            outputStream = socket.getOutputStream();
        } catch (IOException e) {
            return false;
        }
        bufferedOutputStream = new BufferedOutputStream(outputStream);
        InputStreamReader inputStreamReader = new InputStreamReader(inputStream);
        bufferedReader = new BufferedReader(inputStreamReader);
        return true;
    }

    public void close() {
        try {
            bufferedReader.close();
            bufferedOutputStream.close();
        } catch (IOException e) {
            System.out.println("关闭失败");
        }
    }

    public boolean sendMessage(String message) {
        try {
            bufferedOutputStream.write(message.getBytes(StandardCharsets.UTF_8));
            bufferedOutputStream.flush();
        } catch (IOException e) {
            System.out.println("连接已断开");
            return false;
        }
        return true;
    }

    public String receiveMessage() {
        try {
            return bufferedReader.readLine();
        } catch (IOException e) {
            return null;
        }
    }
}

import java.io.IOException;
import java.net.ServerSocket;
import java.net.Socket;
import java.util.concurrent.TimeUnit;

public class Server {
    public static void main(String[] args) throws IOException {
        ServerSocket serverSocket = new ServerSocket(8080);
        Socket socket = serverSocket.accept();
        // 这里只模拟单线程
        Connect connect = new Connect(socket);
        boolean alive = connect.start();
        while (alive) {
            boolean b = connect.sendMessage("我是服务端\n");
            if (!b) break;
            String s = connect.receiveMessage();
            System.out.println(s);
            try {
                TimeUnit.SECONDS.sleep(5);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }
}

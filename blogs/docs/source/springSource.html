# 1.AbstractBeanFactory

public abstract class AbstractBeanFactory extends FactoryBeanRegistrySupport implements ConfigurableBeanFactory {

    /**
     * Names of beans that are currently in creation
     * 当前正在创建的原型bean. 
     */
    private final ThreadLocal<Object> prototypesCurrentlyInCreation = new NamedThreadLocal<>("Prototype beans currently in creation");
    // 该方法用于获取bean
    protected <T> T doGetBean(String name, @Nullable Class<T> requiredType, @Nullable Object[] args, boolean typeCheckOnly) {
        // 官方注释:Eagerly check singleton cache for manually registered singletons.
        // 获取单例bean，非单例bean无法获取
        Object sharedInstance = getSingleton(beanName);
        // 原型bean是否正在创建
        isPrototypeCurrentlyInCreation(beanName);
        // 获取标准beanName
        String nameToLookup = originalBeanName(name);
        if (parentBeanFactory instanceof AbstractBeanFactory) {
            // 从Bean工厂的父工厂获取
            return ((AbstractBeanFactory) parentBeanFactory).doGetBean(nameToLookup, requiredType, args, typeCheckOnly);
        }
        // typeCheckOnly为true表示doGetBean仅仅是为了做类型检查
        // false表示为了实际使用bean实例
        if (!typeCheckOnly) {
            // 需要实际使用bean实例标识为已创建（RootBeanDefinition.stale = true）
            markBeanAsCreated(beanName);
        }
        // dep是beanName的依赖，查看dep是否依赖beanName（即循环依赖），有则抛出异常
        isDependent(beanName, dep);
        // 依赖注册
        registerDependentBean(dep, beanName);
        // 根据依赖关系获取全部依赖的bean
        getBean(dep);
        // Create bean instance.
        if (mbd.isSingleton()) {
            sharedInstance = getSingleton(beanName, () -> {
                try {
                    return createBean(beanName, mbd, args);
                }
                catch (BeansException ex) {
                    // Explicitly remove instance from singleton cache: It might have been put there
                    // eagerly by the creation process, to allow for circular reference resolution.
                    // Also remove any beans that received a temporary reference to the bean.
                    // 创建失败的话会删除单例bean以及所有依赖该单例bean的单例bean
                    destroySingleton(beanName);
                    throw ex;
                }
            });
            // 获取bean对象
            beanInstance = getObjectForBeanInstance(sharedInstance, name, beanName, mbd);
        }
        // 类型转换，将beanInstance转换为requiredType类型，name不起实际作用
        return adaptBeanInstance(name, beanInstance, requiredType);
    }

    protected String originalBeanName(String name) {
        // 转化后的beanName（本名），name可能为本名也可能是别名
        String beanName = transformedBeanName(name);
        // 如果是工厂bean在本名前加上工厂类标志FACTORY_BEAN_PREFIX
        if (name.startsWith(FACTORY_BEAN_PREFIX)) {
            beanName = FACTORY_BEAN_PREFIX + beanName;
        }
        return beanName;
    }

    /**
     * BeanFactoryUtils.transformedBeanName(name):非工厂类，即beanName不以BeanFactory.FACTORY_BEAN_PREFIX开头的话直接返回，
     *  工厂类，从transformedBeanNameCache查找对应缓存并删除缓存BeanFactory.FACTORY_BEAN_PREFIX开头
     */
    protected String transformedBeanName(String name) {
        // 以转化过的name作为别名查找本名
        // 别名和本名对应关系保存在SimpleAliasRegistry的aliasMap
        // private final Map<String, String> aliasMap = new ConcurrentHashMap<>(16);
        return canonicalName(BeanFactoryUtils.transformedBeanName(name));
    }
}

public class DefaultSingletonBeanRegistry extends SimpleAliasRegistry implements SingletonBeanRegistry {
    // DefaultSingletonBeanRegistry内有三个缓存map,通过earlySingletonObjects存储提前引用正在创建中的bean解决循环依赖问题
    /** 一级缓存:Cache of singleton objects: bean name to bean instance. */
    private final Map<String, Object> singletonObjects = new ConcurrentHashMap<>(256);
    /** 三级缓存:Cache of singleton factories: bean name to ObjectFactory. */
    private final Map<String, ObjectFactory<?>> singletonFactories = new HashMap<>(16);
    /** 二级缓存:Cache of early singleton objects: bean name to bean instance. */
    private final Map<String, Object> earlySingletonObjects = new ConcurrentHashMap<>(16);
    
    /**
     * Map between dependent bean names: bean name to Set of dependent bean names. 
     * Set依赖key
     */
    private final Map<String, Set<String>> dependentBeanMap = new ConcurrentHashMap<>(64);
    /**
     * Map between depending bean names: bean name to Set of bean names for the bean's dependencies. 
     * key依赖Set
     */
    private final Map<String, Set<String>> dependenciesForBeanMap = new ConcurrentHashMap<>(64);
    
    /**
     * 官方注释说明了通过earlySingletonObjects提前引用正在创建中的bean解决循环依赖问题:
     * <p>Return the (raw) singleton object registered under the given name.
     * <p>Checks already instantiated singletons and also allows for an early
     * reference to a currently created singleton (resolving a circular reference).
     */
    protected Object getSingleton(String beanName, boolean allowEarlyReference) {
        Object singletonObject = this.singletonObjects.get(beanName);
        // 该单例bean是否正在创建
        if (singletonObject == null && isSingletonCurrentlyInCreation(beanName)) {
            singletonObject = this.earlySingletonObjects.get(beanName);
            // 单例bean是否允许提前创建
            if (singletonObject == null && allowEarlyReference) {
                // 尝试获取缓存中的bean失败后,会进入下面的创建bean的部分
                synchronized (this.singletonObjects) {
                    // 官方注释:Consistent creation of early reference within full singleton lock
                    // 重复一遍检查
                    singletonObject = this.singletonObjects.get(beanName);
                    if (singletonObject == null) {
                        singletonObject = this.earlySingletonObjects.get(beanName);
                        if (singletonObject == null) {
                            ObjectFactory<?> singletonFactory = this.singletonFactories.get(beanName);
                            if (singletonFactory != null) {
                                singletonObject = singletonFactory.getObject();
                                // 先放入earlySingletonObjects中,为了让循环依赖时可以提前引用
                                this.earlySingletonObjects.put(beanName, singletonObject);
                                this.singletonFactories.remove(beanName);
                            }
                        }
                    }
                }
            }
        }
    }

    /**
     * 依赖注册
     */
    public void registerDependentBean(String beanName, String dependentBeanName) {
        String canonicalName = canonicalName(beanName);
        synchronized (this.dependentBeanMap) {
            Set<String> dependentBeans = this.dependentBeanMap.computeIfAbsent(canonicalName, k -> new LinkedHashSet<>(8));
            // 添加dependentBeanName对beanName的依赖
            if (!dependentBeans.add(dependentBeanName)) {
                return;
            }
        }
        // 如果beanName和dependentBeanName之前没有依赖关系，则将依赖关系更新到dependenciesForBeanMap
        synchronized (this.dependenciesForBeanMap) {
            Set<String> dependenciesForBean = this.dependenciesForBeanMap.computeIfAbsent(dependentBeanName, k -> new LinkedHashSet<>(8));
            dependenciesForBean.add(canonicalName);
        }
    }
}

# 2.mysql

docker pull mysql:8.0.27

# --name指定容器名字 -v目录挂载 -p指定端口映射 -e设置mysql参数 -d后台运行
# linux中的/mydata/mysql/log对应容器中的/var/log/mysql
docker run -p 3306:3306 --name mysql \
-v /mydata/mysql/log:/var/log \
-v /mydata/mysql/data:/var/lib/mysql \
-v /mydata/mysql/files:/var/lib/mysql-files \
-v /mydata/mysql/conf:/etc/mysql \
-e MYSQL_ROOT_PASSWORD=root \
-e "TZ=Asia/Shanghai" \
-d mysql:8.0.27

# 上面命令可能会无法启动容器
# 1.彻底卸载原先系统中与mysql相关的内容
# 2.清空挂载文件夹

# 进入容器内部,查看有没有生成my.conf
docker exec -it mysql /bin/bash
cd /etc/mysql/
cat my.conf

# 3.redis

docker pull redis
# 非完整的启动命令
docker run --name some-redis -d redis
# 在虚拟机中创建文件,如果直接挂载的话docker会以为挂载的是一个目录
mkdir -p /opt/redis/conf
touch /opt/redis/conf/redis.conf
# 完整的启动命令,-d:加载配置文件,--:redis配置
docker run -p 6379:6379 --name redis \
-v /opt/redis/data:/data \
-v /opt/redis/conf:/etc/redis/redis.conf \
-d redis:6.2.6 redis-server /etc/redis/redis.conf \
-e "TZ=Asia/Shanghai" \
--bind 0.0.0.0 \
--timeout 1000 \
--tcp-keepalive 300 \
--databases 1 \
--save 3600 1 \
--stop-writes-on-bgsave-error yes \
--rdbcompression yes \
--rdbchecksum yes \
--dbfilename dump.rdb \
--rdb-del-sync-files no \
--repl-diskless-load disabled \
--appendonly yes \
--appendfilename "appendonly.aof" \
--appendfsync everysec \
--no-appendfsync-on-rewrite no \
--aof-use-rdb-preamble yes

# 进入redis客户端
docker exec -it redis redis-cli
# 设置redis容器在docker启动的时候启动
docker update redis --restart=always

# 4.nginx

docker pull nginx:1.10
# 随便启动一个nginx实例，只是为了复制出配置文件，放到docker里作为镜像的统一配置
docker run -p 80:80 -p 443:443 --name nginx -d nginx:1.10
# 复制出配置文件
docker container cp nginx:/etc/nginx /opt/nginx
mv /opt/nginx/ /opt/nginx/conf
# 停掉nginx
docker stop nginx
docker rm nginx

# 创建最终的nginx
# 需要设置时区，否则nginx日志输出时间不正确
docker run -p 80:80 -p 443:443 --name nginx --restart=always \
-v /opt/nginx/html:/usr/share/nginx/html \
-v /opt/nginx/logs:/var/log/nginx \
-v /opt/nginx/conf:/etc/nginx \
-e "TZ=Asia/Shanghai" \
-d nginx:1.10
# 注意一下这个路径映射到了/usr/share/nginx/html，我们在nginx配置文件中是写/usr/share/nginx/html，不是写/opt/nginx/html

# 5.ElasticSearch

# 1.安装ElasticSearch

# 9200是用户交互端口,发送rest api请求的端口; 9300是集群心跳端口通信端口
# -e指定是单阶段运行
docker run --name elasticsearch --restart=always -p 9200:9200 -p 9300:9300 \
-e "discovery.type=single-node" \
-e ES_JAVA_OPTS="-Xms512m -Xmx1024m" \
-e http.host=0.0.0.0 \
-e xpack.security.enabled=false \
-e "TZ=Asia/Shanghai" \
-v /opt/elasticsearch/plugins:/usr/share/elasticsearch/plugins \
-d elasticsearch:8.4.3

# 宿主机目录
# /opt/elasticsearch/config
# /opt/elasticsearch/data
# 容器目录
# config目录: /usr/share/elasticsearch/config/
# data目录: /usr/share/elasticsearch/data/

# 如果添加selinux规则，将要挂载的目录添加到白名单
chmod -R 777 /opt/elasticsearch/plugins

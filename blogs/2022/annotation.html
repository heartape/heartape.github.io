# 2.定义注解和处理器

语法树的api较为晦涩难懂，看看就行

@Target({ElementType.METHOD})
// 需要在编译时改变字节码
@Retention(RetentionPolicy.SOURCE)
public @interface Getter {
}

// 所有注解的全名称
@SupportedAnnotationTypes({"com.heartape.annotation.Getter"})
// jdk环境为java8
@SupportedSourceVersion(SourceVersion.RELEASE_8)
@AutoService(Processor.class)
public class RedisRoProcessor extends AbstractProcessor {
    // 编译期打log
    private Messager messager;

    @Override
    public synchronized void init(ProcessingEnvironment processingEnv) {
        super.init(processingEnv);
        // 获取工具
        this.messager = processingEnv.getMessager();
    }

    @Override
    public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {

        messager.printMessage(Diagnostic.Kind.NOTE, "into processed");
        // 使用注解的程序在idea进行编译时可能导致此处类型转换错误,需要一下设置,原因是idea对虚拟机进行了修改
        // Settings -> Compiler -> Shared build process VM options: -Djps.track.ap.dependencies=false
        Context context = ((JavacProcessingEnvironment) processingEnv).getContext();
        // 提供了待处理的抽象语法树
        JavacTrees trees = JavacTrees.instance(context);
        // 封装了创建AST节点的一些方法
        TreeMaker treeMaker = TreeMaker.instance(context);
        // 提供了创建标识符的方法
        Names names = Names.instance(context);

        // 利用roundEnv的getElementsAnnotatedWith方法过滤出被Getter这个注解标记的类，并存入set
        Set<? extends Element> set = roundEnv.getElementsAnnotatedWith(Getter.class);
        // 遍历这个set里的每一个元素，并生成jCTree这个语法树
        set.forEach(element -> {
            JCTree jcTree = trees.getTree(element);
            // 创建一个TreeTranslator，并重写其中的visitClassDef方法，这个方法处理遍历语法树得到的类定义部分jcClassDecl
            // 利用创建的TreeTranslator去处理jcTree
            jcTree.accept(new TreeTranslator() {
                @Override
                public void visitClassDef(JCTree.JCClassDecl jcClassDecl) {
                    // 创建一个jcVariableDeclList保存类的成员变量
                    List<JCTree.JCVariableDecl> jcVariableDeclList = List.nil();
                    // 遍历jcTree的所有成员(包括成员变量和成员函数和构造函数)，过滤出其中的成员变量，并添加进jcVariableDeclList
                    for (JCTree tree : jcClassDecl.defs) {
                        if (tree.getKind().equals(Tree.Kind.VARIABLE)) {
                            JCTree.JCVariableDecl jcVariableDecl = (JCTree.JCVariableDecl) tree;
                            jcVariableDeclList = jcVariableDeclList.append(jcVariableDecl);
                        }
                    }
                    // 将jcVariableDeclList的所有变量转换成需要添加的getter方法，并添加进jcClassDecl的成员中
                    jcVariableDeclList.forEach(jcVariableDecl -> {
                        messager.printMessage(Diagnostic.Kind.NOTE, jcVariableDecl.getName() + " has been processed");
                        jcClassDecl.defs = jcClassDecl.defs.prepend(makeGetterMethodDecl(jcVariableDecl));
                    });
                    // 调用默认的遍历方法遍历处理后的jcClassDecl
                    super.visitClassDef(jcClassDecl);
                }
                
            });
        });
        return true;
    }

    private JCTree.JCMethodDecl makeGetterMethodDecl(JCTree.JCVariableDecl jcVariableDecl) {

        ListBuffer<JCTree.JCStatement> statements = new ListBuffer<>();
        statements.append(treeMaker.Return(treeMaker.Select(treeMaker.Ident(names.fromString("this")), jcVariableDecl.getName())));
        JCTree.JCBlock body = treeMaker.Block(0, statements.toList());
        return treeMaker.MethodDef(treeMaker.Modifiers(Flags.PUBLIC), getNewMethodName(jcVariableDecl.getName()), jcVariableDecl.vartype, List.nil(), List.nil(), List.nil(), body, null);
    }

    private Name getNewMethodName(Name name) {
        String s = name.toString();
        return names.fromString("get" + s.substring(0, 1).toUpperCase() + s.substring(1, name.length()));
    }
}

public class MyAnnotationBootstrapConfiguration implements ImportBeanDefinitionRegistrar {

    public static final String MY_ANNOTATION_PROCESSOR_BEAN_NAME = "com.heartape.annotation.config.internalMyAnnotationProcessor";

    @Override
    public void registerBeanDefinitions(@Nullable AnnotationMetadata importingClassMetadata, BeanDefinitionRegistry registry) {
        if (!registry.containsBeanDefinition(MY_ANNOTATION_PROCESSOR_BEAN_NAME)) {
            registry.registerBeanDefinition(MY_ANNOTATION_PROCESSOR_BEAN_NAME, new RootBeanDefinition(MyAnnotationBeanPostProcessor.class));
        }
    }

}

# 4.MyAnnotationBeanPostProcessor

public class MyAnnotationBeanPostProcessor implements BeanPostProcessor {
    /**
     * 前处理
     */
    @Override
    public Object postProcessBeforeInitialization(Object bean, String beanName) throws BeansException {
        return bean;
    }

    /**
     * 后处理
     */
    @Override
    public Object postProcessAfterInitialization(Object bean, String beanName) throws BeansException {
        Class<?> targetClass = AopUtils.getTargetClass(bean);
        // 所有的方法和注解信息
        final List<MyAnnotationMethod> methods = new ArrayList<>();
        ReflectionUtils.doWithMethods(targetClass, method -> {
            Collection<MyAnnotation> listenerAnnotations = findListenerAnnotations(method);
            if (listenerAnnotations.size() > 0) {
                methods.add(new MyAnnotationMethod(method, listenerAnnotations.toArray(new MyAnnotation[listenerAnnotations.size()])));
            }
        }, ReflectionUtils.USER_DECLARED_METHODS);
        for (MyAnnotationMethod method : methods) {
            for (MyAnnotation myAnnotation : method.annotations) {
                process(targetClass, method.method, myAnnotation);
            }
        }
        // 不对原bean进行修改
        return bean;
    }

    /**
     * 实现一个简单的功能:启动时执行一次方法
     */
    private void process(Class<?> targetClass, Method method, MyAnnotation myAnnotation) {
        try {
            Object instance = targetClass.getDeclaredConstructor().newInstance();
            method.invoke(instance, myAnnotation.value());
        } catch (InstantiationException | IllegalAccessException | InvocationTargetException | NoSuchMethodException e) {
            e.printStackTrace();
        }
    }

    private Collection<MyAnnotation> findListenerAnnotations(AnnotatedElement element) {
        return MergedAnnotations.from(element, MergedAnnotations.SearchStrategy.TYPE_HIERARCHY)
                .stream(MyAnnotation.class)
                .map(ann -> ann.synthesize())
                .collect(Collectors.toList());
    }

    private static class MyAnnotationMethod {

        final Method method;

        final MyAnnotation[] annotations;

        MyAnnotationMethod(Method method, MyAnnotation[] annotations) {
            this.method = method;
            this.annotations = annotations;
        }
    }
}
